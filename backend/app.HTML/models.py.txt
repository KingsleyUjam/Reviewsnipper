"""
models.py
─────────────────────────────────────────────────────────────────────────────
SQLAlchemy ORM models for ReviewSnipper.

Every table in the PostgreSQL database has a corresponding class here.
Alembic reads these models to generate migration scripts automatically.

Current models:
    User — registered platform users

Models added in later stages:
    Business          — Stage 3
    Competitor        — Stage 3
    PlatformConnection — Stage 3
    RawReview         — Stage 4
    AnalysisSnapshot  — Stage 5
─────────────────────────────────────────────────────────────────────────────
"""

from datetime import datetime, timezone

from sqlalchemy import (
    Boolean,
    Column,
    DateTime,
    Float,
    String,
    text,
)
from sqlalchemy.dialects.postgresql import UUID

from .database import Base


# ── User ──────────────────────────────────────────────────────────────────────

class User(Base):
    """
    A registered ReviewSnipper user.

    One row per user account.  Tracks identity, authentication credentials,
    their primary brand profile, subscription status, and trial period.

    Location fields (target_brand_lat / target_brand_long)
    ───────────────────────────────────────────────────────
    Store the latitude/longitude of the user's primary business location.
    Used by the competitor search feature to find nearby rivals via the
    Google Places API — works globally, no postcode/zip dependency.

    Subscription lifecycle
    ──────────────────────
    new user  →  'free_tier'      (permanent free plan)
    on trial  →  'trial'          (14-day Pro trial, auto-started on register)
    paid      →  'starter' | 'smb_pro' | 'agency'
    expired   →  'free_tier'      (downgraded when trial ends or payment fails)
    cancelled →  'cancelled'      (access ends at period end)
    """

    __tablename__ = "users"

    # ── Primary key ──────────────────────────────────────────────────────────
    # UUID generated by PostgreSQL's gen_random_uuid() function.
    # Using server_default keeps ID generation inside the database —
    # it is never generated by Python, which prevents ID collisions
    # in concurrent requests.
    id = Column(
        UUID(as_uuid=True),
        primary_key=True,
        server_default=text("gen_random_uuid()"),
        nullable=False,
        comment="Unique user identifier — generated by PostgreSQL",
    )

    # ── Identity ──────────────────────────────────────────────────────────────
    # Unique index on email for fast lookup during login.
    # Stored in lowercase — enforced at the application layer in /auth/register
    # and /auth/login to prevent duplicate accounts with different casing.
    email = Column(
        String(320),         # RFC 5321 maximum email length
        unique=True,
        nullable=False,
        index=True,
        comment="User email address — stored lowercase, unique per account",
    )

    # ── Authentication ────────────────────────────────────────────────────────
    # bcrypt hash of the user's password.
    # The plaintext password is NEVER stored — only this hash.
    # passlib generates and verifies bcrypt hashes.
    hashed_password = Column(
        String(72),          # bcrypt output is always 60 chars; 72 gives headroom
        nullable=False,
        comment="bcrypt hash of the user's password — never store plaintext",
    )

    # ── Brand profile ─────────────────────────────────────────────────────────
    # The primary business the user is tracking.
    # Set during the onboarding flow after registration.
    # Nullable — not required at registration time.
    target_brand_name = Column(
        String(255),
        nullable=True,
        comment="Name of the user's primary tracked business",
    )

    # Latitude coordinate of the primary business location.
    # Used for local competitor radius search via Google Places API.
    # Global format — works for any location worldwide.
    # Example: 51.5074 (London), 40.7128 (New York), -33.8688 (Sydney)
    target_brand_lat = Column(
        Float,
        nullable=True,
        comment="Latitude of the user's primary business — for local competitor search",
    )

    # Longitude coordinate of the primary business location.
    # Example: -0.1278 (London), -74.0060 (New York), 151.2093 (Sydney)
    target_brand_long = Column(
        Float,
        nullable=True,
        comment="Longitude of the user's primary business — for local competitor search",
    )

    # ── Subscription ─────────────────────────────────────────────────────────
    # Tracks the user's current plan.
    # Updated by Stripe webhook events (payment_succeeded, subscription_updated,
    # subscription_deleted) processed in the /webhooks/stripe endpoint.
    #
    # Valid values:
    #   free_tier  — permanent free plan (default)
    #   trial      — 14-day Pro trial (auto-started on registration)
    #   starter    — £29/month
    #   smb_pro    — £79/month
    #   agency     — £149/month (coming soon)
    #   cancelled  — subscription cancelled, access ends at period end
    subscription_status = Column(
        String(32),
        nullable=False,
        default="free_tier",
        server_default="free_tier",
        comment="Current subscription plan — updated by Stripe webhooks",
    )

    # When the 14-day trial expires.
    # NULL for users on permanent free_tier with no trial.
    # When this timestamp passes without a paid subscription,
    # a Celery task downgrades subscription_status back to 'free_tier'.
    trial_ends_at = Column(
        DateTime(timezone=True),
        nullable=True,
        comment="Trial expiry timestamp — NULL if no trial was started",
    )

    # Stripe customer ID — set when the user first starts a paid subscription.
    # Used to link ReviewSnipper users to Stripe customer objects.
    # NULL until the user upgrades from free.
    stripe_customer_id = Column(
        String(64),
        nullable=True,
        unique=True,
        comment="Stripe customer ID — set on first paid subscription",
    )

    # Whether the user has verified their email address.
    # Unverified users can browse but cannot run analysis.
    # Set to True when the user clicks the verification link in their
    # welcome email.
    is_verified = Column(
        Boolean,
        nullable=False,
        default=False,
        server_default="false",
        comment="True once the user clicks the email verification link",
    )

    # ── Timestamps ────────────────────────────────────────────────────────────
    # server_default=text("NOW()") means PostgreSQL sets this on INSERT —
    # Python does not need to supply a value.
    # timezone=True stores the timestamp with UTC timezone information.
    created_at = Column(
        DateTime(timezone=True),
        nullable=False,
        server_default=text("NOW()"),
        comment="UTC timestamp when the account was created",
    )

    updated_at = Column(
        DateTime(timezone=True),
        nullable=False,
        server_default=text("NOW()"),
        onupdate=datetime.now(timezone.utc),
        comment="UTC timestamp of the last account update",
    )

    # ── String representation for debugging ──────────────────────────────────
    def __repr__(self) -> str:
        return (
            f"<User id={self.id} email={self.email} "
            f"subscription={self.subscription_status}>"
        )

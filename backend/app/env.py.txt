"""
alembic/env.py
─────────────────────────────────────────────────────────────────────────────
Alembic migration environment configuration.

This file runs every time an Alembic command is executed.
It connects Alembic to your database and your SQLAlchemy models.

Key points:
- DATABASE_URL is read from the environment (not hardcoded)
- All models are imported so Alembic can detect schema changes
- Both online (live DB) and offline (generate SQL only) modes supported
─────────────────────────────────────────────────────────────────────────────
"""

import os
import sys
from logging.config import fileConfig
from pathlib import Path

from alembic import context
from dotenv import load_dotenv
from sqlalchemy import engine_from_config, pool

# ── Make app importable from the alembic/env.py context ──────────────────────
# Alembic runs from the backend/ directory, so we add it to sys.path
# so that "from app.database import Base" works correctly.
sys.path.insert(0, str(Path(__file__).resolve().parents[1]))

# ── Load environment variables ────────────────────────────────────────────────
load_dotenv()

# ── Import Base and all models ────────────────────────────────────────────────
# IMPORTANT: Every model module must be imported here.
# Alembic uses Base.metadata to detect which tables exist and what changed.
# If a model is not imported here, Alembic will not detect changes to it.
from app.database import Base       # noqa: F401 — needed for Base.metadata
from app.models import User         # noqa: F401 — registers User table

# Import additional models here as they are created in later stages:
# from app.models_business import Business, Competitor, PlatformConnection
# from app.models_reviews import RawReview, AnalysisSnapshot

# ── Alembic configuration object ─────────────────────────────────────────────
config = context.config

# Set up Python logging from the alembic.ini [loggers] section
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

# ── Inject DATABASE_URL from environment ─────────────────────────────────────
# This overrides the empty sqlalchemy.url in alembic.ini.
# The URL comes from your .env file (local) or Railway variables (production).
database_url = os.environ.get("DATABASE_URL", "")

if not database_url:
    raise RuntimeError(
        "DATABASE_URL environment variable is not set.\n"
        "Alembic needs a database connection to generate or apply migrations."
    )

# Fix Railway's 'postgres://' prefix
if database_url.startswith("postgres://"):
    database_url = database_url.replace("postgres://", "postgresql://", 1)

config.set_main_option("sqlalchemy.url", database_url)

# ── Target metadata for autogenerate ─────────────────────────────────────────
# Alembic compares Base.metadata (your model definitions) against the live
# database schema to determine what has changed and needs migration.
target_metadata = Base.metadata


# ── Migration modes ───────────────────────────────────────────────────────────

def run_migrations_offline() -> None:
    """
    Run migrations in 'offline' mode.

    Generates SQL scripts without connecting to the database.
    Useful for reviewing migration SQL before applying it.

    Run with: alembic upgrade head --sql
    """
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
        compare_type=True,           # detect column type changes
        compare_server_default=True, # detect default value changes
    )
    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online() -> None:
    """
    Run migrations in 'online' mode — connects to the live database.

    This is the normal mode used when running:
        alembic upgrade head
        alembic downgrade -1
    """
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,    # don't pool connections in migration context
    )

    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            compare_type=True,
            compare_server_default=True,
        )
        with context.begin_transaction():
            context.run_migrations()


# ── Run in the appropriate mode ───────────────────────────────────────────────
if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
